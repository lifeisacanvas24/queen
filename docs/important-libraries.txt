## **Script 1: NSE Data Fetcher (using nsetools)**

This script extracts complete instrument panel data including 52-week high/low from NSE.

```python
#!/usr/bin/env python3
"""
NSE Stock Data Fetcher
Extracts comprehensive instrument data including 52-week high/low from NSE India
"""

import time
from datetime import datetime
from nsetools import Nse
from NSEDownload import stocks
import pandas as pd
import json

class NSEDataFetcher:
    def __init__(self, delay=1):
        """
        Initialize NSE fetcher with rate limiting

        Args:
            delay (int): Delay in seconds between API calls (default: 1)
        """
        self.nse = Nse()
        self.delay = delay

    def get_stock_quote(self, symbol):
        """
        Fetch complete stock quote with 52-week high/low and all instrument fields

        Args:
            symbol (str): NSE stock symbol (e.g., 'RELIANCE', 'INFY')

        Returns:
            dict: Complete stock data including 52w fields
        """
        try:
            # Validate symbol first
            if not self.nse.is_valid_code(symbol):
                print(f"‚ùå Invalid NSE symbol: {symbol}")
                return None

            print(f"Fetching NSE data for {symbol}...")
            data = self.nse.get_quote(symbol)

            # Extract key instrument panel fields
            instrument_data = {
                # Basic Info
                'symbol': data.get('symbol'),
                'company_name': data.get('companyName'),
                'isin_code': data.get('isinCode'),
                'series': data.get('series'),

                # Price Data
                'last_price': data.get('lastPrice'),
                'previous_close': data.get('previousClose'),
                'open_price': data.get('open'),
                'day_high': data.get('dayHigh'),
                'day_low': data.get('dayLow'),

                # 52-Week High/Low (Primary fields)
                '52_week_high': data.get('high52'),
                '52_week_low': data.get('low52'),

                # Additional 52-Week Data
                'week_high_low': data.get('weekHighLow', {}),

                # Volume & Turnover
                'total_traded_volume': data.get('totalTradedVolume'),
                'total_traded_value': data.get('totalTradedValue'),
                'quantity_traded': data.get('quantityTraded'),

                # Market Depth
                'market_cap': data.get('marketCap'),
                'free_float_market_cap': data.get('ffmc'),

                # Dates
                'last_update_time': data.get('lastUpdateTime'),
                'ex_date': data.get('exDate'),
                'record_date': data.get('recordDate'),
                'bc_start_date': data.get('bcStartDate'),
                'bc_end_date': data.get('bcEndDate'),

                # Corporate Actions
                ' Dividend': data.get('dividend'),
                'bonus': data.get('bonus'),
                'right': data.get('right'),
                'book_closure_start': data.get('bcStartDate'),
                'book_closure_end': data.get('bcEndDate'),

                # Technical Indicators
                'price_band_upper': data.get('pricebandupper'),
                'price_band_lower': data.get('pricebandlower'),
                'var_margin': data.get('varMargin'),

                # Delivery Data
                'delivery_quantity': data.get('deliveryQuantity'),
                'delivery_to_traded_quantity': data.get('deliveryToTradedQuantity'),

                # Metadata
                'face_value': data.get('faceValue'),
                'security_type': data.get('securityType'),
                'price_hint': data.get('priceband'),

                # Raw data for debugging
                '_raw': data
            }

            time.sleep(self.delay)  # Rate limiting
            return instrument_data

        except Exception as e:
            print(f"‚ùå Error fetching {symbol}: {str(e)}")
            return None

    def get_historical_with_52week(self, symbol, start_date, end_date):
        """
        Fetch historical data with 52-week high/low columns

        Args:
            symbol (str): Stock symbol
            start_date (str): Start date (DD-MM-YYYY)
            end_date (str): End date (DD-MM-YYYY)

        Returns:
            pd.DataFrame: DataFrame with 52-week high/low columns
        """
        try:
            print(f"Fetching historical data for {symbol}...")

            df = stocks.get_data(
                stock_symbol=symbol,
                start_date=start_date,
                end_date=end_date
            )

            # Ensure 52-week columns are present
            if '52 Week High Price' in df.columns and '52 Week Low Price' in df.columns:
                df['52_week_high'] = df['52 Week High Price']
                df['52_week_low'] = df['52 Week Low Price']

            return df

        except Exception as e:
            print(f"‚ùå Error fetching historical data: {str(e)}")
            return None

    def get_52_week_extremes(self):
        """
        Fetch all stocks currently at 52-week highs or lows

        Returns:
            dict: Stocks at 52-week highs and lows
        """
        try:
            highs = self.nse.get_52_week_high()
            lows = self.nse.get_52_week_low()

            return {
                '52_week_highs': highs,
                '52_week_lows': lows
            }
        except Exception as e:
            print(f"‚ùå Error fetching extremes: {str(e)}")
            return None

def main():
    """Example usage of NSE data fetcher"""

    # Initialize fetcher with 1-second delay
    fetcher = NSEDataFetcher(delay=1)

    # Example 1: Get complete quote for a single stock
    print("\n" + "="*60)
    print("EXAMPLE 1: Single Stock Quote with 52-Week Data")
    print("="*60)

    symbol = "RELIANCE"
    quote = fetcher.get_stock_quote(symbol)

    if quote:
        print(f"\nüìä Instrument Panel for {symbol}")
        print("-" * 40)
        print(f"Company: {quote['company_name']}")
        print(f"Last Price: ‚Çπ{quote['last_price']}")
        print(f"Day High/Low: ‚Çπ{quote['day_high']} / ‚Çπ{quote['day_low']}")
        print(f"52-Week High: ‚Çπ{quote['52_week_high']}")
        print(f"52-Week Low: ‚Çπ{quote['52_week_low']}")
        print(f"Total Volume: {quote['total_traded_volume']:,}")
        print(f"Market Cap: ‚Çπ{quote['market_cap']:,}")
        print(f"ISIN Code: {quote['isin_code']}")

        # Save to JSON
        with open(f'{symbol}_nse_data.json', 'w') as f:
            json.dump(quote, f, indent=2, default=str)
        print(f"\n‚úÖ Data saved to {symbol}_nse_data.json")

    # Example 2: Get historical data with 52-week bands
    print("\n" + "="*60)
    print("EXAMPLE 2: Historical Data with 52-Week Bands")
    print("="*60)

    hist_df = fetcher.get_historical_with_52week(
        symbol="TCS",
        start_date="01-01-2024",
        end_date="31-01-2024"
    )

    if hist_df is not None:
        print(f"\nüìà Historical Data (first 5 rows):")
        print(hist_df.head())
        print(f"\nColumns available: {list(hist_df.columns)}")

        # Save to CSV
        hist_df.to_csv('TCS_historical_52week.csv', index=False)
        print("\n‚úÖ Historical data saved to TCS_historical_52week.csv")

    # Example 3: Get stocks at 52-week extremes
    print("\n" + "="*60)
    print("EXAMPLE 3: Stocks at 52-Week Extremes")
    print("="*60)

    extremes = fetcher.get_52_week_extremes()

    if extremes:
        print(f"\nüî∫ Stocks at 52-Week Highs: {len(extremes['52_week_highs'])}")
        for stock in extremes['52_week_highs'][:5]:  # Top 5
            print(f"  - {stock['symbol']}: ‚Çπ{stock['ltp']} (High: ‚Çπ{stock['new52WHL']})")

        print(f"\nüîª Stocks at 52-Week Lows: {len(extremes['52_week_lows'])}")
        for stock in extremes['52_week_lows'][:5]:  # Top 5
            print(f"  - {stock['symbol']}: ‚Çπ{stock['ltp']} (Low: ‚Çπ{stock['new52WHL']})")

if __name__ == "__main__":
    print("NSE Stock Data Fetcher - Starting...")
    main()
```

## **Script 2: BSE/NSE Unified Data Fetcher (using Indian Stock Market API)**

This script fetches data for both NSE and BSE stocks, including BSE-only symbols like NSDL, using the free unified API.

```python
#!/usr/bin/env python3
"""
Unified Indian Stock Market Data Fetcher
Supports both NSE and BSE exchanges via free API
Handles BSE-only symbols like NSDL
"""

import requests
import json
import time
from datetime import datetime
import sys

class IndianStockMarketAPI:
    """
    Free API for fetching NSE & BSE stock data
    Base URL: http://nse-api-khaki.vercel.app:5000
    """

    def __init__(self, base_url="http://nse-api-khaki.vercel.app:5000", delay=0.5):
        """
        Initialize API client

        Args:
            base_url (str): API base URL
            delay (float): Delay between requests in seconds
        """
        self.base_url = base_url.rstrip('/')
        self.delay = delay
        self.session = requests.Session()
        self.session.headers.update({
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
        })

    def _make_request(self, endpoint, params=None):
        """
        Make API request with error handling

        Args:
            endpoint (str): API endpoint
            params (dict): Query parameters

        Returns:
            dict: JSON response or None
        """
        try:
            url = f"{self.base_url}{endpoint}"
            response = self.session.get(url, params=params, timeout=10)
            response.raise_for_status()

            time.sleep(self.delay)  # Rate limiting

            return response.json()

        except requests.exceptions.RequestException as e:
            print(f"‚ùå API Request Error: {str(e)}")
            return None
        except json.JSONDecodeError as e:
            print(f"‚ùå JSON Decode Error: {str(e)}")
            return None

    def get_stock_details(self, symbol, exchange=None, response_format='num'):
        """
        Fetch detailed stock data for NSE or BSE symbol

        Args:
            symbol (str): Stock symbol (e.g., 'RELIANCE', 'NSDL')
            exchange (str): 'NSE' or 'BSE' (auto-detected if None)
            response_format (str): 'num' for numbers, 'val' for values with units

        Returns:
            dict: Complete stock details including 52-week data
        """
        # Auto-detect exchange for BSE-only symbols
        if exchange is None:
            if symbol.upper() in ['NSDL', 'CDSL', 'MCX', 'NCDEX']:
                exchange = 'BSE'
                symbol_with_suffix = f"{symbol}.BO"
            else:
                symbol_with_suffix = symbol
        else:
            suffix = '.NS' if exchange.upper() == 'NSE' else '.BO'
            symbol_with_suffix = f"{symbol}{suffix}" if not symbol.endswith(suffix) else symbol

        print(f"Fetching data for {symbol_with_suffix}...")

        params = {'symbol': symbol_with_suffix, 'res': response_format}
        data = self._make_request('/stock', params)

        if not data or data.get('status') != 'success':
            print(f"‚ùå Failed to fetch data for {symbol}")
            return None

        stock_data = data.get('data', {})

        # Extract comprehensive fields
        instrument_data = {
            # Symbol Info
            'symbol': stock_data.get('symbol'),
            'ticker': stock_data.get('ticker'),
            'company_name': stock_data.get('company_name'),
            'exchange': stock_data.get('exchange'),

            # Price Data
            'current_price': stock_data.get('current_price'),
            'previous_close': stock_data.get('previous_close'),
            'open_price': stock_data.get('open_price'),
            'day_high': stock_data.get('day_high'),
            'day_low': stock_data.get('day_low'),

            # 52-Week High/Low
            '52_week_high': stock_data.get('week_52_high'),
            '52_week_low': stock_data.get('week_52_low'),

            # Volume Data
            'volume': stock_data.get('volume'),
            'total_traded_value': stock_data.get('total_traded_value'),

            # Market Data
            'market_cap': stock_data.get('market_cap'),
            'free_float_market_cap': stock_data.get('free_float_market_cap'),
            'free_float_shares': stock_data.get('free_float_shares'),

            # Valuation
            'pe_ratio': stock_data.get('pe_ratio'),
            'pb_ratio': stock_data.get('pb_ratio'),
            'dividend_yield': stock_data.get('dividend_yield'),
            'face_value': stock_data.get('face_value'),
            'book_value': stock_data.get('book_value'),
            'eps': stock_data.get('eps'),

            # Price Bands
            'price_band_upper': stock_data.get('price_band_upper'),
            'price_band_lower': stock_data.get('price_band_lower'),
            'upper_band': stock_data.get('upper_band'),
            'lower_band': stock_data.get('lower_band'),

            # Corporate Actions
            'dividend': stock_data.get('dividend'),
            'bonus': stock_data.get('bonus'),
            'ex_date': stock_data.get('ex_date'),
            'record_date': stock_data.get('record_date'),
            'bc_start_date': stock_data.get('bc_start_date'),
            'bc_end_date': stock_data.get('bc_end_date'),

            # Metadata
            'isin_code': stock_data.get('isin_code'),
            'security_code': stock_data.get('security_code'),
            'industry': stock_data.get('industry'),
            'sector': stock_data.get('sector'),
            'tick_size': stock_data.get('tick_size'),
            'lot_size': stock_data.get('lot_size'),

            # Timestamps
            'last_updated': stock_data.get('last_updated'),
            'timestamp': stock_data.get('timestamp'),

            # Additional Info
            'smart_advice': stock_data.get('smart_advice'),
            'price_change': stock_data.get('price_change'),
            'price_change_pct': stock_data.get('price_change_pct'),
            'p_change': stock_data.get('pChange'),
            'change': stock_data.get('change'),

            # Raw response for debugging
            '_raw': stock_data
        }

        return instrument_data

    def get_multiple_stocks(self, symbols, response_format='num'):
        """
        Fetch data for multiple stocks in one request

        Args:
            symbols (list): List of stock symbols
            response_format (str): 'num' or 'val'

        Returns:
            list: List of stock data dictionaries
        """
        symbols_str = ','.join(symbols)
        params = {'symbols': symbols_str, 'res': response_format}

        data = self._make_request('/stock/list', params)

        if not data or data.get('status') != 'success':
            print("‚ùå Failed to fetch multiple stocks")
            return None

        return data.get('data', [])

    def search_symbol(self, query):
        """
        Search for stock symbols by company name

        Args:
            query (str): Search query (e.g., 'reliance', 'nsdl')

        Returns:
            list: Search results with symbols
        """
        params = {'q': query}
        data = self._make_request('/search', params)

        if not data or data.get('status') != 'success':
            print(f"‚ùå Search failed for '{query}'")
            return None

        return data.get('data', [])

    def get_all_symbols(self):
        """
        Get all available cached symbols with NSE and BSE tickers

        Returns:
            dict: Available symbols
        """
        data = self._make_request('/symbols')

        if not data or data.get('status') != 'success':
            print("‚ùå Failed to fetch symbols list")
            return None

        return data.get('data', {})
```

## **NSE API - 52 Week High/Low Availability**

**Yes, NSE APIs do provide 52-week high/low fields.** The data is available through multiple endpoints:

### **1. Stock Quote API (with all_data=True)**
The NSETools library's `get_quote()` method returns 52-week data in the `weekHighLow` object:
```json
{
  "weekHighLow": {
    "min": 4890,  // 52-week low
    "max": 5218.45 // 52-week high
  }
}
```

### **2. Index Stock Quotes**
When fetching stocks within an index, the API directly provides `yearHigh` and `yearLow` fields:

## **For BSE-Only Listed Symbols (like NSDL)**

- **Third-party aggregators**: APIs that scrape or aggregate BSE data

### **Limitations & Considerations**

- **No free official BSE API**: Unlike NSE, BSE doesn't provide a free public API
- **Data latency**: Third-party APIs may have 15-20 minute delay
- **Rate limits**: Third-party services have usage limits
- **Authentication**: Most require API key registration
- **Coverage**: Verify the API covers your specific BSE-only symbols

### **Alternative: Web Scraping**

**Bottom Line**: For BSE-only symbols, you'll need to rely on third-party aggregators or commercial data vendors, as BSE doesn't offer a public API comparable to NSE's.
